# Установка зависимостей для Telegram-бота BSG

## 1. Создание и активация виртуального окружения (рекомендуется)
python3 -m venv .venv
source .venv/bin/activate

## 2. Обновление менеджера пакетов pip
python -m pip install --upgrade pip

## 3. Установка основных библиотек бота одной командой
pip install aiogram==2.25.1 openpyxl pillow requests
